h3. {color:#36b37e}ETAPE 3 : Ajout des balises Gestion{color} {status:colour=Green|title=INSERT}

||Action||Ajouter les balises Gestion (ExtraitReglement) pour BBNP75804-EUR||

||Responsable|| {status:colour=Purple|title=TechLead} ||

{code:sql|title=SQL - INSERT}
INSERT INTO TA_RN_BALISE_PAR_COMPTE (
    ID_COMPTE_ACCURATE,
    NUM_COL_FEEDER,
    ID_BALISE,
    INDICATEUR_MANUEL_AUTO
)
SELECT
    (SELECT ID_COMPTE_ACCURATE
     FROM TA_RN_COMPTE_ACCURATE
     WHERE NUM_COMPTE_ACCURATE = 'BBNP75804-EUR') AS ID_COMPTE_ACCURATE,
    BalRef.NUM_COL_FEEDER,
    BalRef.ID_BALISE,
    BalRef.INDICATEUR_MANUEL_AUTO
FROM TA_RN_BALISE_PAR_COMPTE BalRef
LEFT JOIN TA_RN_BALISE_PAR_COMPTE Balise1
    ON Balise1.NUM_COL_FEEDER = BalRef.NUM_COL_FEEDER
    AND Balise1.ID_BALISE = BalRef.ID_BALISE
    AND Balise1.INDICATEUR_MANUEL_AUTO = BalRef.INDICATEUR_MANUEL_AUTO
    AND Balise1.ID_COMPTE_ACCURATE IN (
        SELECT ID_COMPTE_ACCURATE
        FROM TA_RN_COMPTE_ACCURATE
        WHERE NUM_COMPTE_ACCURATE = 'BBNP75804-EUR'
    )
WHERE BalRef.ID_COMPTE_ACCURATE IN (
    SELECT ID_COMPTE_ACCURATE
    FROM TA_RN_COMPTE_ACCURATE
    WHERE NUM_COMPTE_ACCURATE = 'BBNP40692-EUR'
)
AND BalRef.ID_BALISE IN (
    SELECT ID_BALISE
    FROM TA_RN_BALISE
    WHERE Type_BALISE = 'GEST'
)
-- Condition pour n'avoir que les nouvelles balises
AND Balise1.NUM_COL_FEEDER IS NULL
ORDER BY BalRef.NUM_COL_FEEDER;
{code}

(i) *Note :* La condition {{Balise1.NUM_COL_FEEDER IS NULL}} garantit l'insertion uniquement des nouvelles balises.
