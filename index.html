Script d'analyse - Recherche UTL_FILE dans les scripts 
#!/bin/bash
# Script d'audit UTL_FILE dans les scripts SQL
# Usage: ./audit_utl_file.sh /chemin/scripts

SCRIPTS_DIR="${1:-.}"
OUTPUT_FILE="audit_utl_file_$(date +%Y%m%d_%H%M%S).csv"

echo "Script,Ligne,Occurrence" > "$OUTPUT_FILE"

find "$SCRIPTS_DIR" -type f \( -name "*.sql" -o -name "*.pls" -o -name "*.pkb" -o -name "*.pks" \) | while read script; do
    grep -n -i "UTL_FILE" "$script" 2>/dev/null | while read match; do
        line_num=$(echo "$match" | cut -d: -f1)
        content=$(echo "$match" | cut -d: -f2-)
        echo "\"$script\",$line_num,\"$content\"" >> "$OUTPUT_FILE"
    done
done

echo "Audit terminé. Résultat: $OUTPUT_FILE"
echo "Nombre de scripts avec UTL_FILE: $(tail -n +2 "$OUTPUT_FILE" | cut -d, -f1 | sort -u | wc -l)"

🔍 Script d'audit UTL_FILE (schéma connecté)

⚡
Exécute ce script dans SQL Developer
Il analyse automatiquement le schéma sur lequel tu es connecté. Pas besoin de modifier le script !
SQL - Audit UTL_FILE du schéma connecté 
-- ============================================
-- AUDIT UTL_FILE - SCHÉMA CONNECTÉ
-- ============================================
-- Exécute ce script connecté à ton schéma applicatif dans SQL Developer
-- user_source = code source du schéma sur lequel tu es connecté

-- 1. Afficher le schéma courant
SELECT USER AS schema_courant FROM DUAL;


-- 2. Nombre total de références UTL_FILE dans ce schéma
SELECT COUNT(*) AS total_references_utl_file
FROM user_source
WHERE UPPER(text) LIKE '%UTL_FILE%';


-- 3. Détail : quels objets utilisent UTL_FILE ?
SELECT
    name AS object_name,
    type AS object_type,
    COUNT(*) AS nb_lignes
FROM user_source
WHERE UPPER(text) LIKE '%UTL_FILE%'
GROUP BY name, type
ORDER BY nb_lignes DESC;


-- 4. Voir le code exact (lignes contenant UTL_FILE) avec Directory utilisé
-- Fonctions UTL_FILE avec directory en paramètre : FOPEN, FOPEN_NCHAR, FREMOVE, FCOPY, FRENAME, FGETATTR
SELECT
    name AS object_name,
    type AS object_type,
    line AS numero_ligne,
    -- Extraction du directory (1er paramètre des fonctions UTL_FILE qui utilisent un directory)
    NVL(
        -- Directory en chaîne littérale 'DIR_NAME'
        REGEXP_SUBSTR(text, 'UTL_FILE\.(FOPEN|FOPEN_NCHAR|FREMOVE|FCOPY|FRENAME|FGETATTR)\s*\(\s*''([^'']+)''', 1, 1, 'i', 2),
        -- Directory en variable PL/SQL
        REGEXP_SUBSTR(text, 'UTL_FILE\.(FOPEN|FOPEN_NCHAR|FREMOVE|FCOPY|FRENAME|FGETATTR)\s*\(\s*([A-Za-z_][A-Za-z0-9_]*)', 1, 1, 'i', 2)
    ) AS directory_utilise,
    text AS code
FROM user_source
WHERE UPPER(text) LIKE '%UTL_FILE%'
ORDER BY name, type, line;


-- ============================================
-- 5. LISTER TOUTES LES FONCTIONS/CONSTANTES UTL_FILE UTILISÉES
-- ============================================
-- Cette requête liste TOUTES les fonctions et constantes UTL_FILE
-- utilisées dans ton code (sans filtre, pour ne rien omettre)

COLUMN fonction_utl_file FORMAT A40
COLUMN nb_utilisations FORMAT 9999

SELECT
    REGEXP_SUBSTR(UPPER(text), 'UTL_FILE\.[A-Z_]+') AS fonction_utl_file,
    COUNT(*) AS nb_utilisations
FROM user_source
WHERE UPPER(text) LIKE '%UTL_FILE.%'
GROUP BY REGEXP_SUBSTR(UPPER(text), 'UTL_FILE\.[A-Z_]+')
ORDER BY nb_utilisations DESC;

-- Fonctions supportées par le wrapper :
-- ✅ FOPEN, FCLOSE, FCLOSE_ALL, PUT_LINE, PUT, GET_LINE, NEW_LINE
-- ✅ FFLUSH, IS_OPEN, PUT_RAW, GET_RAW, FGETATTR, FCOPY, FRENAME, FREMOVE
-- ✅ Exceptions : INVALID_PATH, INVALID_MODE, INVALID_FILEHANDLE, etc.
-- ❌ Non supportées : FSEEK, FGETPOS, PUTF


-- ============================================
-- 6. INVESTIGUER UNE FONCTION UTL_FILE INCONNUE
-- ============================================
-- Si tu vois une fonction UTL_FILE que tu ne connais pas dans la liste,
-- utilise cette requête pour voir où elle est utilisée dans le code
-- Remplace 'FILE_OPEN' par la fonction à investiguer

SELECT
    name AS object_name,
    type AS object_type,
    line AS numero_ligne,
    text AS code
FROM user_source
WHERE UPPER(text) LIKE '%FILE_OPEN%'
ORDER BY name, type, line;

SQL - Lister tous les schémas utilisant UTL_FILE (SYSDBA) 
-- ============================================
-- LISTER LES SCHÉMAS UTILISANT UTL_FILE
-- ============================================
-- Exécute connecté en SYSDBA (sqlplus / as sysdba)

-- Formatage pour affichage lisible dans SQL*Plus
SET LINESIZE 200
SET PAGESIZE 100
COLUMN schema_name FORMAT A30
COLUMN nb_references FORMAT 999999

-- 1. Liste simple des schémas
SELECT DISTINCT owner AS schema_name
FROM dba_source
WHERE UPPER(text) LIKE '%UTL_FILE%'
  AND owner NOT IN ('SYS', 'SYSTEM', 'ORACLE_OCM', 'XDB', 'WMSYS')
ORDER BY owner;


-- 2. Avec le nombre de références par schéma
SELECT owner AS schema_name, COUNT(*) AS nb_references
FROM dba_source
WHERE UPPER(text) LIKE '%UTL_FILE%'
  AND owner NOT IN ('SYS', 'SYSTEM', 'ORACLE_OCM', 'XDB', 'WMSYS')
GROUP BY owner
ORDER BY nb_references DESC;

FROM dba_users

    'SYS', 'SYSTEM', 'ORACLE_OCM', 'XDB', 'WMSYS', 'CTXSYS', 'MDSYS',
    'OLAPSYS', 'ORDDATA', 'ORDSYS', 'OUTLN', 'DBSNMP', 'APPQOSSYS',

💡 En résumé simple (pour expliquer comme un pro) :

🐧 Actuellement : Oracle sur serveurs Linux On-Premise

Vos bases Oracle sont installées sur des serveurs Linux classiques dans votre datacenter. Vous gérez tout : le matériel (serveurs, stockage, réseau), le système d'exploitation Linux, l'installation d'Oracle, les patchs de sécurité, les sauvegardes, etc. C'est un modèle CAPEX (investissement) avec un contrôle total mais une charge opérationnelle importante.

☁️ Cible : ExaCC (Exadata Cloud@Customer)

Exadata est une appliance Oracle : du matériel spécialement conçu et optimisé pour faire tourner Oracle Database (stockage flash, réseau InfiniBand, Smart Scan, etc.). C'est la Rolls-Royce des infrastructures Oracle.

Cloud@Customer signifie qu'Oracle installe cette Exadata physiquement dans votre datacenter, mais c'est Oracle qui la gère (maintenance, patchs, monitoring matériel).

Vous passez en modèle OPEX (abonnement) et vous vous concentrez uniquement sur vos bases de données (schémas, données, PL/SQL). Vos données restent chez vous = souveraineté des données garantie.

🎯 La différence clé en une phrase :

"On passe de serveurs Linux classiques qu'on gère nous-mêmes à une infrastructure Exadata managée par Oracle, installée chez nous. On gagne en performance, en fiabilité et on réduit notre charge opérationnelle, tout en gardant nos données dans notre datacenter."

⚠️ Impact technique majeur pour le DBA :

Sur ExaCC, tu n'as plus d'accès au filesystem du serveur (pas de SSH sur /u01/app/...). Les packages comme UTL_FILE qui écrivent sur le disque local ne fonctionnent plus.

Tu dois utiliser Object Storage (stockage objet OCI) via le package DBMS_CLOUD pour toutes les opérations fichiers.

👉 C'est exactement l'objet de cette migration : adapter ton code PL/SQL pour qu'il fonctionne sur ExaCC.

🔒 C'est quoi l'ACL exactement ?

🔐
ACL = Access Control List (Liste de Contrôle d'Accès)

Même si ta base ExaCC et Object Storage sont tous les deux dans le cloud OCI, ils ne peuvent pas se parler automatiquement. C'est une mesure de sécurité.

L'ACL est une règle qui dit : "La base de données ExaCC a le droit de se connecter à Object Storage sur le port 443 (HTTPS)"

❌ Sans ACL configurée

Ta base essaie d'accéder à Object Storage → BLOQUÉ par le firewall interne OCI

Erreur : ORA-29024: Certificate validation failure ou timeout

✅ Avec ACL configurée

Ta base accède à Object Storage → AUTORISÉ

Les fichiers sont lus/écrits normalement

🎯 Bonne nouvelle pour toi :

L'ACL est configurée par les admins OCI, pas par toi. Tu dois juste leur demander de confirmer que c'est fait. Dans ton email, demande simplement : "Pouvez-vous confirmer que l'ACL autorise la base ExaCC à accéder à Object Storage ?"

SQL - Liste des fonctions UTL_FILE utilisées et leur support 
-- ============================================
-- FONCTIONS UTL_FILE UTILISÉES ET COMPATIBILITÉ WRAPPER
-- ============================================
-- Cette requête liste toutes les fonctions UTL_FILE utilisées
-- et indique si elles sont supportées par UTL_FILE_WRAPPER

SET LINESIZE 200
SET PAGESIZE 100
COLUMN fonction FORMAT A20
COLUMN nb_utilisations FORMAT 9999
COLUMN supporte_wrapper FORMAT A20

SELECT
    fonction,
    nb_utilisations,
    CASE
        WHEN fonction IN ('FOPEN', 'FCLOSE', 'FCLOSE_ALL', 'PUT_LINE', 'PUT',
                          'GET_LINE', 'NEW_LINE', 'FFLUSH', 'IS_OPEN',
                          'PUT_RAW', 'GET_RAW', 'FGETATTR',
                          'FCOPY', 'FRENAME', 'FREMOVE', 'FILE_TYPE')
        THEN '✅ OUI'
        ELSE '❌ NON'
    END AS supporte_wrapper
FROM (
    SELECT
        CASE
            WHEN UPPER(text) LIKE '%UTL_FILE.FOPEN%' THEN 'FOPEN'
            WHEN UPPER(text) LIKE '%UTL_FILE.FCLOSE_ALL%' THEN 'FCLOSE_ALL'
            WHEN UPPER(text) LIKE '%UTL_FILE.FCLOSE%' THEN 'FCLOSE'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT_LINE%' THEN 'PUT_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT_RAW%' THEN 'PUT_RAW'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT%' AND UPPER(text) NOT LIKE '%PUT_LINE%' AND UPPER(text) NOT LIKE '%PUT_RAW%' THEN 'PUT'
            WHEN UPPER(text) LIKE '%UTL_FILE.GET_LINE%' THEN 'GET_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.GET_RAW%' THEN 'GET_RAW'
            WHEN UPPER(text) LIKE '%UTL_FILE.NEW_LINE%' THEN 'NEW_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FFLUSH%' THEN 'FFLUSH'
            WHEN UPPER(text) LIKE '%UTL_FILE.IS_OPEN%' THEN 'IS_OPEN'
            WHEN UPPER(text) LIKE '%UTL_FILE.FGETATTR%' THEN 'FGETATTR'
            WHEN UPPER(text) LIKE '%UTL_FILE.FCOPY%' THEN 'FCOPY'
            WHEN UPPER(text) LIKE '%UTL_FILE.FRENAME%' THEN 'FRENAME'
            WHEN UPPER(text) LIKE '%UTL_FILE.FREMOVE%' THEN 'FREMOVE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FILE_TYPE%' OR UPPER(text) LIKE '%FILE_TYPE%' THEN 'FILE_TYPE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FSEEK%' THEN 'FSEEK'
            WHEN UPPER(text) LIKE '%UTL_FILE.FGETPOS%' THEN 'FGETPOS'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUTF%' THEN 'PUTF'
            ELSE 'AUTRE'
        END AS fonction,
        COUNT(*) AS nb_utilisations
    FROM user_source
    WHERE UPPER(text) LIKE '%UTL_FILE%'
    GROUP BY
        CASE
            WHEN UPPER(text) LIKE '%UTL_FILE.FOPEN%' THEN 'FOPEN'
            WHEN UPPER(text) LIKE '%UTL_FILE.FCLOSE_ALL%' THEN 'FCLOSE_ALL'
            WHEN UPPER(text) LIKE '%UTL_FILE.FCLOSE%' THEN 'FCLOSE'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT_LINE%' THEN 'PUT_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT_RAW%' THEN 'PUT_RAW'
            WHEN UPPER(text) LIKE '%UTL_FILE.PUT%' AND UPPER(text) NOT LIKE '%PUT_LINE%' AND UPPER(text) NOT LIKE '%PUT_RAW%' THEN 'PUT'
            WHEN UPPER(text) LIKE '%UTL_FILE.GET_LINE%' THEN 'GET_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.GET_RAW%' THEN 'GET_RAW'
            WHEN UPPER(text) LIKE '%UTL_FILE.NEW_LINE%' THEN 'NEW_LINE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FFLUSH%' THEN 'FFLUSH'
            WHEN UPPER(text) LIKE '%UTL_FILE.IS_OPEN%' THEN 'IS_OPEN'
            WHEN UPPER(text) LIKE '%UTL_FILE.FGETATTR%' THEN 'FGETATTR'

            WHEN UPPER(text) LIKE '%UTL_FILE.FRENAME%' THEN 'FRENAME'
            WHEN UPPER(text) LIKE '%UTL_FILE.FREMOVE%' THEN 'FREMOVE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FILE_TYPE%' OR UPPER(text) LIKE '%FILE_TYPE%' THEN 'FILE_TYPE'
            WHEN UPPER(text) LIKE '%UTL_FILE.FSEEK%' THEN 'FSEEK'

Avantage Explication

🔄 Rollback complet si problème

⚠️
Script de rollback complet
Ce script annule entièrement la migration et revient à UTL_FILE natif Oracle. À exécuter en tant que DBA (SYSDBA).
SQL - ROLLBACK COMPLET (exécuter en SYSDBA) 
-- ============================================
-- ROLLBACK COMPLET - MIGRATION UTL_FILE
-- ============================================
-- Exécuter en tant que SYSDBA
-- Ce script annule toute la migration et revient à UTL_FILE natif

SET SERVEROUTPUT ON
SET LINESIZE 200

DECLARE
    v_count NUMBER;
    v_schema VARCHAR2(128) := 'SCHEMA_WRAPPER';  -- ⚠️ ADAPTER: Schéma où le wrapper a été installé
BEGIN
    DBMS_OUTPUT.PUT_LINE('========================================');
    DBMS_OUTPUT.PUT_LINE('ROLLBACK MIGRATION UTL_FILE - DÉBUT');
    DBMS_OUTPUT.PUT_LINE('========================================');
    DBMS_OUTPUT.PUT_LINE('');

    -- ============================================
    -- ÉTAPE 1 : Supprimer le synonyme PUBLIC
    -- ============================================
    DBMS_OUTPUT.PUT_LINE('ÉTAPE 1 : Suppression du synonyme PUBLIC UTL_FILE...');
    BEGIN
        EXECUTE IMMEDIATE 'DROP PUBLIC SYNONYM UTL_FILE';
        DBMS_OUTPUT.PUT_LINE('   ✅ Synonyme PUBLIC UTL_FILE supprimé');
    EXCEPTION
        WHEN OTHERS THEN
            IF SQLCODE = -1432 THEN
                DBMS_OUTPUT.PUT_LINE('   ℹ️ Synonyme PUBLIC UTL_FILE n''existait pas');
            ELSE
                DBMS_OUTPUT.PUT_LINE('   ❌ Erreur: ' || SQLERRM);
            END IF;
    END;

    -- ============================================
    -- ÉTAPE 2 : Supprimer le package UTL_FILE_WRAPPER
    -- ============================================
    DBMS_OUTPUT.PUT_LINE('');
    DBMS_OUTPUT.PUT_LINE('ÉTAPE 2 : Suppression du package UTL_FILE_WRAPPER...');
    BEGIN
        EXECUTE IMMEDIATE 'DROP PACKAGE ' || v_schema || '.UTL_FILE_WRAPPER';
        DBMS_OUTPUT.PUT_LINE('   ✅ Package UTL_FILE_WRAPPER supprimé');
    EXCEPTION
        WHEN OTHERS THEN
            IF SQLCODE = -4043 THEN
                DBMS_OUTPUT.PUT_LINE('   ℹ️ Package UTL_FILE_WRAPPER n''existait pas');
            ELSE
                DBMS_OUTPUT.PUT_LINE('   ❌ Erreur: ' || SQLERRM);
            END IF;
    END;

    -- ============================================
    -- ÉTAPE 3 : Supprimer la table de mapping
    -- ============================================
    DBMS_OUTPUT.PUT_LINE('');
    DBMS_OUTPUT.PUT_LINE('ÉTAPE 3 : Suppression de la table de mapping...');
    BEGIN
        EXECUTE IMMEDIATE 'DROP TABLE ' || v_schema || '.UTL_FILE_DIRECTORY_MAPPING CASCADE CONSTRAINTS';
        DBMS_OUTPUT.PUT_LINE('   ✅ Table UTL_FILE_DIRECTORY_MAPPING supprimée');
    EXCEPTION
        WHEN OTHERS THEN
            IF SQLCODE = -942 THEN
                DBMS_OUTPUT.PUT_LINE('   ℹ️ Table UTL_FILE_DIRECTORY_MAPPING n''existait pas');
            ELSE
                DBMS_OUTPUT.PUT_LINE('   ❌ Erreur: ' || SQLERRM);
            END IF;
    END;

    -- ============================================
    -- ÉTAPE 4 : Recompiler tous les schémas impactés
    -- ============================================
    DBMS_OUTPUT.PUT_LINE('');
    DBMS_OUTPUT.PUT_LINE('ÉTAPE 4 : Recompilation des schémas impactés...');
    FOR rec IN (
        SELECT DISTINCT owner
        FROM dba_source
        WHERE UPPER(text) LIKE '%UTL_FILE%'
          AND owner NOT IN ('SYS', 'SYSTEM', 'ORACLE_OCM', 'XDB', 'WMSYS')
        ORDER BY owner
    ) LOOP
        BEGIN
            DBMS_OUTPUT.PUT_LINE('   Recompilation de ' || rec.owner || '...');
            DBMS_UTILITY.COMPILE_SCHEMA(schema => rec.owner, compile_all => FALSE);
            DBMS_OUTPUT.PUT_LINE('   ✅ ' || rec.owner || ' recompilé');
        EXCEPTION
            WHEN OTHERS THEN
                DBMS_OUTPUT.PUT_LINE('   ❌ Erreur recompilation ' || rec.owner || ': ' || SQLERRM);
        END;
    END LOOP;

    -- ============================================
    -- ÉTAPE 5 : Vérification finale
    -- ============================================
    DBMS_OUTPUT.PUT_LINE('');
    DBMS_OUTPUT.PUT_LINE('ÉTAPE 5 : Vérifications...');

    -- Vérifier qu'il n'y a plus de synonyme PUBLIC UTL_FILE
    SELECT COUNT(*) INTO v_count
    FROM dba_synonyms
    WHERE synonym_name = 'UTL_FILE' AND owner = 'PUBLIC';

    IF v_count = 0 THEN
        DBMS_OUTPUT.PUT_LINE('   ✅ Aucun synonyme PUBLIC UTL_FILE');
    ELSE
        DBMS_OUTPUT.PUT_LINE('   ❌ Synonyme PUBLIC UTL_FILE existe encore !');
    END IF;

    -- Vérifier les objets invalides
    SELECT COUNT(*) INTO v_count
    FROM dba_objects
    WHERE status = 'INVALID'
      AND owner NOT IN ('SYS', 'SYSTEM', 'ORACLE_OCM');

    IF v_count = 0 THEN
        DBMS_OUTPUT.PUT_LINE('   ✅ Aucun objet invalide');
    ELSE
        DBMS_OUTPUT.PUT_LINE('   ⚠️ ' || v_count || ' objets invalides (vérifier avec requête ci-dessous)');
    END IF;

    DBMS_OUTPUT.PUT_LINE('');
    DBMS_OUTPUT.PUT_LINE('========================================');
    DBMS_OUTPUT.PUT_LINE('ROLLBACK TERMINÉ');
    DBMS_OUTPUT.PUT_LINE('Le code utilise maintenant UTL_FILE natif Oracle (SYS.UTL_FILE)');
    DBMS_OUTPUT.PUT_LINE('========================================');
END;
/

        credential_name     => 'CRED_Q08449A'
    );
END;

1. Deploiement du Wallet Oracle

🔐
OBLIGATOIRE : Le Wallet est indispensable pour la connexion mTLS sur ExaCC (port 1522). Sans Wallet, aucune connexion possible !

Qu'est-ce que le Wallet ?

C'est un dossier securise contenant des fichiers de securite pour se connecter a la base de donnees cloud :

Contenu du dossier Wallet 
📁 Wallet_MonExaCC/
   ├── cwallet.sso      ← Certificats SSL (le plus important)
   ├── tnsnames.ora     ← Noms des services de connexion
   ├── sqlnet.ora       ← Configuration reseau
   └── ewallet.p12      ← Cle privee (format PKCS12)

Pourquoi c'est obligatoire sur ExaCC ?

Element On-Premise (Avant) ExaCC (Apres)
Type de connexion Connexion simple TCP Connexion chiffree SSL/TLS
Port 1521 1522 (mTLS)
Certificat Pas de certificat Certificat obligatoire
Commande connexion sqlplus user/pwd@host:1521/service sqlplus user/pwd@service_high (via Wallet)

Analogie simple

💡
Imagine le Wallet comme une carte d'acces securisee :

Avant (on-premise) : Tu donnais juste ton nom et mot de passe a la porte
Apres (ExaCC) : Tu dois presenter ta carte magnetique (le Wallet) + ton mot de passe

Sans la carte (Wallet), meme avec le bon mot de passe, la porte reste fermee.

🤝 EXPLICATION DÉTAILLÉE : Le HANDSHAKE SSL/TLS (Étape par Étape)

Le HANDSHAKE est la "poignée de main" entre le client (votre serveur applicatif) et le serveur (ExaCC) pour établir une connexion sécurisée. C'est un processus d'échange de messages qui permet de :

  • ✅ Vérifier l'identité des deux parties (authentification mutuelle)
  • ✅ Négocier les algorithmes de chiffrement à utiliser
  • ✅ Échanger des clés de session pour chiffrer les données
  • ✅ Établir un canal sécurisé avant d'envoyer les données sensibles
📤 PHASE 1 : Client Hello (Le client se présente)

Que fait le client ?

Le serveur applicatif (client) envoie un message "Hello" à ExaCC contenant :

  • Version TLS supportée (ex: TLS 1.2, TLS 1.3)
  • Liste des suites de chiffrement (cipher suites) supportées
    Ex: TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384
  • Nombre aléatoire (Client Random) - utilisé plus tard pour générer les clés
  • Session ID (pour réutiliser une session existante si possible)
  • Extensions (SNI - Server Name Indication, etc.)
📨 Message envoyé : "Bonjour ExaCC ! Je supporte TLS 1.2/1.3, voici mes algorithmes..."
📥 PHASE 2 : Server Hello (ExaCC répond et envoie son certificat)

Que fait ExaCC ?

ExaCC répond avec plusieurs messages :

a) Server Hello

  • Version TLS choisie (celle qui est commune)
  • Suite de chiffrement choisie parmi celles proposées par le client
  • Nombre aléatoire (Server Random)
  • Session ID

b) Certificate (Certificat du serveur)

  • Certificat SSL d'ExaCC signé par une autorité de certification (CA)
  • Contient la clé publique d'ExaCC
  • Contient l'identité d'ExaCC (nom de domaine, organisation, etc.)
  • Permet au client de vérifier l'identité du serveur

c) Certificate Request (Demande de certificat client)

  • ⚠️ Spécifique au mTLS (pas dans TLS simple)
  • ExaCC demande au client d'envoyer son certificat
  • C'est l'authentification mutuelle

d) Server Hello Done

  • Message indiquant que le serveur a fini d'envoyer ses messages
  • C'est maintenant au tour du client de répondre
📨 Message reçu : "J'utilise TLS 1.2 + AES-256. Voici mon certificat. Envoie-moi le tien !"
🔐 PHASE 3 : Client Certificate & Key Exchange (Le client s'authentifie)

Que fait le client ?

a) Vérification du certificat serveur

  • Le client vérifie que le certificat d'ExaCC est valide
  • Vérifie la signature de l'autorité de certification (CA)
  • Vérifie que le nom de domaine correspond
  • Vérifie que le certificat n'est pas expiré
  • Si la vérification échoue → ❌ Connexion refusée

b) Certificate (Envoi du certificat client)

  • Le client envoie son certificat (extrait de cwallet.sso)
  • Contient la clé publique du client
  • Permet à ExaCC de vérifier l'identité du client
  • ⚠️ La clé privée reste sur le serveur applicatif !

c) Client Key Exchange (Échange de clé de session)

  • Le client génère une "Pre-Master Secret" (clé temporaire aléatoire)
  • Chiffre cette clé avec la clé publique d'ExaCC (du certificat serveur)
  • Envoie cette clé chiffrée à ExaCC
  • ✅ Seul ExaCC peut déchiffrer avec sa clé privée
  • Cette clé servira à générer les clés de session symétriques

d) Certificate Verify (Preuve de possession de la clé privée)

  • Le client signe tous les messages précédents du handshake
  • Utilise sa clé privée (ewallet.p12) pour créer la signature
  • Envoie la signature à ExaCC
  • ExaCC vérifie la signature avec la clé publique du certificat client
  • ✅ Prouve que le client possède bien la clé privée correspondant au certificat
  • C'est ici que la clé privée est utilisée (JAMAIS envoyée !)

e) Change Cipher Spec

  • Le client indique qu'il va maintenant utiliser le chiffrement négocié
  • Tous les messages suivants seront chiffrés

f) Finished

  • Message chiffré contenant un hash de tous les messages du handshake
  • Permet de vérifier que le handshake n'a pas été altéré
📨 Message envoyé : "Ton certificat est OK. Voici le mien + preuve que j'ai la clé privée !"
✅ PHASE 4 : Server Finished (ExaCC confirme et établit la connexion)

Que fait ExaCC ?

a) Vérification du certificat client

  • ExaCC vérifie le certificat client reçu
  • Vérifie la signature de la CA
  • Vérifie que le certificat est dans la liste autorisée

b) Vérification de la signature (Certificate Verify)

  • ExaCC utilise la clé publique du certificat client
  • Vérifie la signature numérique envoyée par le client
  • Si valide → ✅ Le client possède bien la clé privée correspondante
  • Si invalide → ❌ Connexion refusée

c) Génération des clés de session

  • ExaCC déchiffre le "Pre-Master Secret" avec sa clé privée
  • Combine : Pre-Master Secret + Client Random + Server Random
  • Génère les clés de session symétriques (AES par exemple)
  • Ces clés seront utilisées pour chiffrer toutes les données de la session

d) Change Cipher Spec

  • ExaCC indique qu'il passe en mode chiffré

e) Finished

  • Message chiffré confirmant que le handshake est terminé
  • Contient un hash de tous les messages pour vérifier l'intégrité
📨 Message reçu : "Ton certificat est valide ! Canal sécurisé établi. Tu peux envoyer tes données."
🎉 HANDSHAKE TERMINÉ - CONNEXION SÉCURISÉE ÉTABLIE

Ce qui a été accompli :

  • ✅ Authentification mutuelle : Les deux parties ont prouvé leur identité
  • ✅ Clés de session générées : Chiffrement symétrique rapide (AES)
  • ✅ Canal sécurisé : Toutes les données seront chiffrées
  • ✅ Intégrité garantie : Impossible de modifier les données en transit

→ Maintenant l'application peut envoyer USER/PASSWORD de manière sécurisée

💡 Point clé du HANDSHAKE :

Le handshake est comme un contrôle de sécurité à l'aéroport :
1️⃣ Vous montrez votre passeport (certificat)
2️⃣ L'agent vérifie que c'est bien le vôtre (signature avec clé privée)
3️⃣ Une fois vérifié, vous pouvez passer (tunnel TLS établi)
4️⃣ Vous ne donnez JAMAIS votre passeport à l'agent, vous le montrez juste !

De même, la clé privée n'est jamais transmise, elle sert uniquement à prouver votre identité en créant une signature que le serveur peut vérifier.

⚠️ CLARIFICATION IMPORTANTE : La clé privée est-elle envoyée au serveur ExaCC ?

❌ NON ! La clé privée NE QUITTE JAMAIS le serveur applicatif

C'est un point crucial de la cryptographie à clé publique (asymétrique). Voici ce qui se passe réellement : 

🔑 CRYPTOGRAPHIE ASYMÉTRIQUE : Clé Publique vs Clé Privée
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📦 Wallet contient 2 éléments distincts :

   1️⃣ CERTIFICAT (dans cwallet.sso)
      ├─ Contient la CLÉ PUBLIQUE du client
      ├─ Contient l'identité du client (DN, organisation, etc.)
      ├─ Signé par une autorité de certification (CA)
      └─ ✅ PEUT être envoyé au serveur (pas de risque)

   2️⃣ CLÉ PRIVÉE (dans ewallet.p12)
      ├─ Reste TOUJOURS sur le serveur applicatif
      ├─ Utilisée UNIQUEMENT pour SIGNER des messages
      ├─ Ne voyage JAMAIS sur le réseau
      └─ ❌ NE DOIT JAMAIS quitter le serveur applicatif
🔐 Comment fonctionne la signature numérique ?

Étape par étape :

  1. Client crée un message
    Message = "Je suis le serveur applicatif X"
  2. Client SIGNE le message avec sa clé privée (ewallet.p12)
    Signature = SIGN(Message, Clé_Privée_Client)
    ⚠️ La clé privée reste sur le serveur applicatif !
  3. Client envoie à ExaCC :
    ➜ Certificat (contient CLÉ PUBLIQUE)
    ➜ Message
    ➜ Signature
    ✅ La clé privée n'est PAS envoyée !
  4. ExaCC VÉRIFIE la signature avec la clé publique du certificat
    VERIFY(Signature, Message, Clé_Publique_Client) == OK ✅
    ℹ️ ExaCC utilise la clé PUBLIQUE (dans le certificat) pour vérifier
  5. Si la vérification réussit :
    ✅ ExaCC sait que le client possède la clé privée correspondante
    ✅ Le client est authentifié
    ✅ La connexion mTLS est établie
🔄 Analogie : Signature manuscrite

Clé Privée = Votre main qui signe un document
Signature = La signature manuscrite sur le papier
Clé Publique = Modèle de votre signature pour vérification

➜ Vous n'envoyez PAS votre main au destinataire !
➜ Vous envoyez seulement le document SIGNÉ
➜ Le destinataire compare votre signature avec le modèle public

-- BODY du package wrapper (simplifié - version complète dans migration-utl-file-exacc.html)
CREATE OR REPLACE PACKAGE BODY EXP_RNAPA.UTL_FILE_WRAPPER AS

    -- Variables privées
    TYPE t_file_content IS TABLE OF VARCHAR2(32767) INDEX BY BINARY_INTEGER;
    TYPE t_file_info IS RECORD (
        content     t_file_content,
        line_count  PLS_INTEGER,
        current_pos PLS_INTEGER,
        mode_char   VARCHAR2(1),
        directory   VARCHAR2(128),
        filename    VARCHAR2(512)
    );
    TYPE t_file_handles IS TABLE OF t_file_info INDEX BY RAW(16);
    g_files t_file_handles;

    -- Obtenir l'URI Object Storage
    FUNCTION get_object_uri(p_directory VARCHAR2, p_filename VARCHAR2) RETURN VARCHAR2 IS
        v_uri VARCHAR2(1000);
    BEGIN
        SELECT object_uri || p_filename INTO v_uri
        FROM EXP_RNAPA.UTL_FILE_DIRECTORY_MAPPING
        WHERE directory_name = UPPER(p_directory)
          AND is_active = 1;
        RETURN v_uri;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RAISE invalid_path;
    END;

    -- FOPEN
    FUNCTION fopen(location IN VARCHAR2, filename IN VARCHAR2, open_mode IN VARCHAR2, max_linesize IN BINARY_INTEGER DEFAULT 1024) RETURN file_type IS
        v_file file_type;
        v_uri VARCHAR2(1000);
        v_content BLOB;
        v_line VARCHAR2(32767);
        v_pos PLS_INTEGER := 1;
    BEGIN
        v_file.id := SYS_GUID();
        v_uri := get_object_uri(location, filename);

        g_files(v_file.id).directory := location;
        g_files(v_file.id).filename := filename;
        g_files(v_file.id).mode_char := UPPER(SUBSTR(open_mode, 1, 1));
        g_files(v_file.id).current_pos := 1;
        g_files(v_file.id).line_count := 0;

        IF g_files(v_file.id).mode_char = 'R' THEN
            -- Lecture depuis Object Storage
            v_content := DBMS_CLOUD.GET_OBJECT(
                credential_name => 'OCI_CREDENTIAL',
                object_uri => v_uri
            );
            -- Parser le contenu ligne par ligne
            DECLARE
                v_clob CLOB := UTL_RAW.CAST_TO_VARCHAR2(v_content);
                v_start PLS_INTEGER := 1;
                v_end PLS_INTEGER;
            BEGIN
                LOOP
                    v_end := INSTR(v_clob, CHR(10), v_start);
                    IF v_end = 0 THEN
                        v_line := SUBSTR(v_clob, v_start);
                        IF LENGTH(v_line) > 0 THEN
                            g_files(v_file.id).line_count := g_files(v_file.id).line_count + 1;
                            g_files(v_file.id).content(g_files(v_file.id).line_count) := RTRIM(v_line, CHR(13));
                        END IF;
                        EXIT;
                    ELSE
                        g_files(v_file.id).line_count := g_files(v_file.id).line_count + 1;
                        g_files(v_file.id).content(g_files(v_file.id).line_count) := RTRIM(SUBSTR(v_clob, v_start, v_end - v_start), CHR(13));
                        v_start := v_end + 1;
                    END IF;
                END LOOP;
            END;
        END IF;

        RETURN v_file;
    END;

    -- IS_OPEN
    FUNCTION is_open(file IN file_type) RETURN BOOLEAN IS
    BEGIN
        RETURN g_files.EXISTS(file.id);
    END;

    -- FCLOSE
    PROCEDURE fclose(file IN OUT file_type) IS
        v_uri VARCHAR2(1000);
        v_content CLOB := '';
    BEGIN
        IF g_files.EXISTS(file.id) THEN
            IF g_files(file.id).mode_char IN ('W', 'A') THEN
                -- Écrire vers Object Storage
                v_uri := get_object_uri(g_files(file.id).directory, g_files(file.id).filename);
                FOR i IN 1..g_files(file.id).line_count LOOP
                    v_content := v_content || g_files(file.id).content(i) || CHR(10);
                END LOOP;
                DBMS_CLOUD.PUT_OBJECT(
                    credential_name => 'OCI_CREDENTIAL',
                    object_uri => v_uri,
                    contents => UTL_RAW.CAST_TO_RAW(v_content)
                );
            END IF;
            g_files.DELETE(file.id);
        END IF;
        file.id := NULL;
    END;

    -- FCLOSE_ALL
    PROCEDURE fclose_all IS
        v_key RAW(16);
    BEGIN
        v_key := g_files.FIRST;
        WHILE v_key IS NOT NULL LOOP
            g_files.DELETE(v_key);
            v_key := g_files.NEXT(v_key);
        END LOOP;
    END;

    -- PUT
    PROCEDURE put(file IN file_type, buffer IN VARCHAR2) IS
    BEGIN
        IF NOT g_files.EXISTS(file.id) THEN
            RAISE invalid_filehandle;
        END IF;
        IF g_files(file.id).line_count = 0 THEN
            g_files(file.id).line_count := 1;
            g_files(file.id).content(1) := buffer;
        ELSE
            g_files(file.id).content(g_files(file.id).line_count) :=
                g_files(file.id).content(g_files(file.id).line_count) || buffer;
        END IF;
    END;

    -- PUT_LINE
    PROCEDURE put_line(file IN file_type, buffer IN VARCHAR2, autoflush IN BOOLEAN DEFAULT FALSE) IS
    BEGIN
        IF NOT g_files.EXISTS(file.id) THEN
            RAISE invalid_filehandle;
        END IF;
        g_files(file.id).line_count := g_files(file.id).line_count + 1;
        g_files(file.id).content(g_files(file.id).line_count) := buffer;
    END;

    -- PUTF
    PROCEDURE putf(file IN file_type, format IN VARCHAR2, arg1 IN VARCHAR2 DEFAULT NULL, arg2 IN VARCHAR2 DEFAULT NULL, arg3 IN VARCHAR2 DEFAULT NULL, arg4 IN VARCHAR2 DEFAULT NULL, arg5 IN VARCHAR2 DEFAULT NULL) IS
        v_result VARCHAR2(32767) := format;
    BEGIN
        v_result := REPLACE(v_result, '%s', NVL(arg1, ''));
        IF arg2 IS NOT NULL THEN v_result := REPLACE(v_result, '%s', arg2); END IF;
        IF arg3 IS NOT NULL THEN v_result := REPLACE(v_result, '%s', arg3); END IF;
        IF arg4 IS NOT NULL THEN v_result := REPLACE(v_result, '%s', arg4); END IF;
        IF arg5 IS NOT NULL THEN v_result := REPLACE(v_result, '%s', arg5); END IF;
        put(file, v_result);
    END;

    -- NEW_LINE
    PROCEDURE new_line(file IN file_type, lines IN NATURAL DEFAULT 1) IS
    BEGIN
        FOR i IN 1..lines LOOP
            put_line(file, '');
        END LOOP;
    END;

    -- GET_LINE
    PROCEDURE get_line(file IN file_type, buffer OUT VARCHAR2, len IN PLS_INTEGER DEFAULT NULL) IS
    BEGIN
        IF NOT g_files.EXISTS(file.id) THEN
            RAISE invalid_filehandle;
        END IF;
        IF g_files(file.id).current_pos > g_files(file.id).line_count THEN
            RAISE NO_DATA_FOUND;
        END IF;
        buffer := g_files(file.id).content(g_files(file.id).current_pos);
        IF len IS NOT NULL AND LENGTH(buffer) > len THEN
            buffer := SUBSTR(buffer, 1, len);
        END IF;
        g_files(file.id).current_pos := g_files(file.id).current_pos + 1;
    END;

    -- FFLUSH
    PROCEDURE fflush(file IN file_type) IS
    BEGIN
        NULL; -- Le flush se fait au FCLOSE
    END;

    -- FREMOVE
    PROCEDURE fremove(location IN VARCHAR2, filename IN VARCHAR2) IS
        v_uri VARCHAR2(1000);
    BEGIN
        v_uri := get_object_uri(location, filename);
        DBMS_CLOUD.DELETE_OBJECT(
            credential_name => 'OCI_CREDENTIAL',
            object_uri => v_uri
        );
    EXCEPTION
        WHEN OTHERS THEN
            RAISE delete_failed;
    END;

    -- FRENAME
    PROCEDURE frename(src_location IN VARCHAR2, src_filename IN VARCHAR2, dest_location IN VARCHAR2, dest_filename IN VARCHAR2, overwrite IN BOOLEAN DEFAULT FALSE) IS
        v_src_uri VARCHAR2(1000);
        v_dest_uri VARCHAR2(1000);
        v_content BLOB;
    BEGIN
        v_src_uri := get_object_uri(src_location, src_filename);
        v_dest_uri := get_object_uri(dest_location, dest_filename);

        v_content := DBMS_CLOUD.GET_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_src_uri);
        DBMS_CLOUD.PUT_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_dest_uri, contents => v_content);
        DBMS_CLOUD.DELETE_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_src_uri);
    EXCEPTION
        WHEN OTHERS THEN
            RAISE rename_failed;
    END;

    -- FCOPY
    PROCEDURE fcopy(src_location IN VARCHAR2, src_filename IN VARCHAR2, dest_location IN VARCHAR2, dest_filename IN VARCHAR2, start_line IN BINARY_INTEGER DEFAULT 1, end_line IN BINARY_INTEGER DEFAULT NULL) IS
        v_src_uri VARCHAR2(1000);
        v_dest_uri VARCHAR2(1000);
        v_content BLOB;
    BEGIN
        v_src_uri := get_object_uri(src_location, src_filename);
        v_dest_uri := get_object_uri(dest_location, dest_filename);

        v_content := DBMS_CLOUD.GET_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_src_uri);
        DBMS_CLOUD.PUT_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_dest_uri, contents => v_content);
    END;

    -- FGETATTR
    FUNCTION fgetattr(location IN VARCHAR2, filename IN VARCHAR2, fexists OUT BOOLEAN, file_length OUT NUMBER, block_size OUT BINARY_INTEGER) RETURN BOOLEAN IS
        v_uri VARCHAR2(1000);
        v_content BLOB;
    BEGIN
        v_uri := get_object_uri(location, filename);
        BEGIN
            v_content := DBMS_CLOUD.GET_OBJECT(credential_name => 'OCI_CREDENTIAL', object_uri => v_uri);
            fexists := TRUE;
            file_length := DBMS_LOB.GETLENGTH(v_content);
            block_size := 4096;
            RETURN TRUE;
        EXCEPTION
            WHEN OTHERS THEN
                fexists := FALSE;
                file_length := 0;
                block_size := 0;
                RETURN FALSE;
        END;
    END;

    -- FGETPOS
    FUNCTION fgetpos(file IN file_type) RETURN PLS_INTEGER IS
    BEGIN
        IF NOT g_files.EXISTS(file.id) THEN
            RAISE invalid_filehandle;
        END IF;
        RETURN g_files(file.id).current_pos;
    END;

    BEGIN
        IF NOT g_files.EXISTS(file.id) THEN
            RAISE invalid_filehandle;
        END IF;
        IF absolute_offset IS NOT NULL THEN
            g_files(file.id).current_pos := absolute_offset;
        ELSIF relative_offset IS NOT NULL THEN
            g_files(file.id).current_pos := g_files(file.id).current_pos + relative_offset;
        END IF;
    END;

🔍 Strategie d'Instrumentation - Detection Code Mort UTL_FILE

Objectif : Identifier les fonctions UTL_FILE reellement utilisees en production avant la migration ExaCC

Date : 05/03/2026

📋 Principe de la Strategie

Avant de migrer les appels UTL_FILE vers DBMS_CLOUD + Object Storage OCI, nous devons identifier :

  • Code Vivant : Fonctions UTL_FILE reellement appelees en production
  • Code Mort : Fonctions UTL_FILE jamais appelees (pas besoin de les migrer)

Approche

  1. Deployer l'infrastructure de logging : Table + Procedure SP_LOG_FICHIER
  2. Instrumenter les packages : Ajouter des appels SP_LOG_FICHIER dans chaque fonction utilisant UTL_FILE
  3. Laisser tourner en production : Pendant quelques semaines/mois
  4. Analyser les logs : Identifier le code mort vs code vivant
  5. Migrer uniquement le code vivant : Gain de temps et reduction des risques
💡 Avantage cle : La procedure SP_LOG_FICHIER utilise PRAGMA AUTONOMOUS_TRANSACTION ce qui garantit que les logs sont ecrits meme si le traitement principal echoue ou fait un ROLLBACK.

CREATE OR REPLACE PROCEDURE SP_LOG_FICHIER (
    p_schema_name       IN VARCHAR2 DEFAULT NULL,
    p_package_name      IN VARCHAR2,
    p_function_name     IN VARCHAR2,
    p_directory_logique IN VARCHAR2 DEFAULT NULL,
    p_directory_oracle  IN VARCHAR2 DEFAULT NULL,
    p_file_name         IN VARCHAR2 DEFAULT NULL,
    p_file_mode         IN VARCHAR2 DEFAULT NULL,  -- FOPEN, APPEND, PUT_LINE, INIT, etc.
    p_contenu_apercu    IN VARCHAR2 DEFAULT NULL,
    p_nb_lignes         IN NUMBER   DEFAULT NULL,
    p_taille_octets     IN NUMBER   DEFAULT NULL,
    p_id_clob           IN NUMBER   DEFAULT NULL,
    p_nom_param         IN VARCHAR2 DEFAULT NULL,
    p_statut            IN VARCHAR2 DEFAULT 'OK',
    p_message_erreur    IN VARCHAR2 DEFAULT NULL
) IS
    PRAGMA AUTONOMOUS_TRANSACTION;  -- Garantit que le log est ecrit meme en cas de ROLLBACK
    v_directory_path VARCHAR2(4000);
    v_db_name        VARCHAR2(128);
    v_sid            NUMBER;
    v_serial         NUMBER;
BEGIN
    -- Recuperer le chemin physique si directory_oracle fourni
    IF p_directory_oracle IS NOT NULL THEN
        BEGIN
            SELECT directory_path INTO v_directory_path
            FROM all_directories
            WHERE directory_name = UPPER(p_directory_oracle);
        EXCEPTION
            WHEN NO_DATA_FOUND THEN v_directory_path := '[DIRECTORY NON TROUVE]';
        END;
    END IF;

    -- Recuperer infos session
    BEGIN
        SELECT name INTO v_db_name FROM v$database;
    EXCEPTION WHEN OTHERS THEN v_db_name := SYS_CONTEXT('USERENV', 'DB_NAME');
    END;

    BEGIN
        SELECT sid, serial# INTO v_sid, v_serial FROM v$session
        WHERE audsid = SYS_CONTEXT('USERENV', 'SESSIONID');
    EXCEPTION WHEN OTHERS THEN v_sid := NULL; v_serial := NULL;
    END;

    -- Inserer le log
    INSERT INTO LOG_FICHIERS_PACKAGES (
        SCHEMA_NAME, PACKAGE_NAME, FUNCTION_NAME, DIRECTORY_LOGIQUE, DIRECTORY_ORACLE,
        DIRECTORY_PATH, FILE_NAME, FILE_MODE, CONTENU_APERCU, NB_LIGNES, TAILLE_OCTETS,
        ID_CLOB, NOM_PARAM, STATUT, MESSAGE_ERREUR, DB_NAME, INSTANCE_NAME, HOST_NAME,
        SERVICE_NAME, OS_USER, IP_ADDRESS, TERMINAL, SID, SERIAL#

Trouver automatiquement tous les comptes avec des rollbacks qui ont encore des records orphelins dans BR_DATA :

SQL - Lister TOUS les rollbacks avec records orphelins 
-- Trouve tous les comptes avec des records orphelins apres rollback

SELECT
    a.acct_id,
    acc.acct_name,
    a.WHICHONE as LOAD_ID_ROLLBACKE,
    TRUNC(a.timestamp) as DATE_ROLLBACK,
    TO_CHAR(a.timestamp, 'HH24:MI:SS') as HEURE,
    d.nb_records as NB_RECORDS_ORPHELINS,
    d.total_amount as TOTAL_AMOUNT
FROM BR_AUDIT a
JOIN (
    -- Sous-requete : compter les records par load_id

    FROM BR_DATA
    GROUP BY acct_id, load_id

LEFT JOIN BS_ACCTS acc ON acc.acct_id = a.acct_id
WHERE a.type = 16  -- EXEC_ROLLBACK
ORDER BY a.timestamp DESC;

📋 Etape 1 : Identifier les Rollbacks (Date Automatique)

Trouver automatiquement tous les rollbacks d'un compte avec les dates et l'ecart potentiel :

SQL - Tous les rollbacks d'un compte (date auto + ecart) 
-- Remplacer UNIQUEMENT :ACCT_ID (pas besoin de date !)
SELECT
    a.WHICHONE as LOAD_ID,
    TO_CHAR(MIN(CASE WHEN a.type = 18 THEN a.timestamp END), 'DD/MM/YYYY HH24:MI:SS') as DEBUT_ROLLBACK,
    TO_CHAR(MAX(CASE WHEN a.type = 16 THEN a.timestamp END), 'DD/MM/YYYY HH24:MI:SS') as EXEC_ROLLBACK,
    (SELECT COUNT(*) FROM BR_DATA d WHERE d.acct_id = :ACCT_ID AND d.load_id = a.WHICHONE) as NB_ORPHELINS,
    (SELECT MAX(amount) - MIN(amount) FROM BR_DATA d WHERE d.acct_id = :ACCT_ID AND d.load_id = a.WHICHONE) as ECART_POTENTIEL
FROM BR_AUDIT a
WHERE a.acct_id = :ACCT_ID
  AND a.type IN (16, 18)
GROUP BY a.acct_id, a.WHICHONE
HAVING (SELECT COUNT(*) FROM BR_DATA d WHERE d.acct_id = :ACCT_ID AND d.load_id = a.WHICHONE) > 0
ORDER BY EXEC_ROLLBACK DESC;
✅
Resultat : Affiche TOUS les rollbacks avec records orphelins, leur date exacte et l'ecart potentiel.
Utilisez le LOAD_ID trouve pour les etapes suivantes.

🎯 Etape 1 bis : Identifier le Load Responsable de l'Ecart

Quand il y a plusieurs rollbacks, cette requete calcule l'ecart NET de chaque load pour trouver celui qui correspond a l'ecart de la Balance Carree :

SQL - Ecart NET par load (identifier le coupable) 
-- Remplacer :ACCT_ID uniquement
-- Calcule l'ecart NET de CHAQUE load orphelin
SELECT
    load_id,
    COUNT(*) as NB_RECORDS,
    SUM(CASE WHEN cs_flag = 'C' THEN amount ELSE 0 END) as TOTAL_COTE_C,
    SUM(CASE WHEN cs_flag = 'S' THEN amount ELSE 0 END) as TOTAL_COTE_S,
    ABS(SUM(CASE WHEN cs_flag = 'C' THEN amount ELSE 0 END) -
        SUM(CASE WHEN cs_flag = 'S' THEN amount ELSE 0 END)) as ECART_NET,
    MAX(trans_date) as DATE_TRANS
FROM BR_DATA d
WHERE d.acct_id = :ACCT_ID
  AND d.load_id IN (
      SELECT DISTINCT a.WHICHONE
      FROM BR_AUDIT a
      WHERE a.acct_id = :ACCT_ID AND a.type = 16
  )
GROUP BY load_id
ORDER BY ECART_NET DESC;
✅
Resultat : Le(s) load(s) avec l'ECART_NET correspondant a l'ecart de la Balance Carree sont les coupables !
Notez les LOAD_ID pour l'Etape 3.

🔄 Exemple Resolu : Compte 1906 - Ecart 2,66€

Objectif : Documentation complete de l'investigation et resolution de l'ecart de 2,66€ sur le compte 1906

Date : 25/02/2026 | Resolution : 08/03/2026

📋 Contexte du Probleme

Lorsqu'un fichier est charge puis annule (rollback), il peut arriver que :

  • Les records du fichier soient supprimes de BR_DATA
  • Mais le solde ne soit pas reinitialise correctement
  • Resultat : ecart dans la Balance Carree
⚠️
Symptome typique : La Balance Carree affiche un ecart (ex: 2,66€) qui ne correspond a aucune transaction visible dans BR_DATA.

🗃️ Tables Cles a Connaitre

Table Role Colonnes Importantes
BR_AUDIT Historique de TOUTES les operations acct_id, record_id, load_id, type, bfr_state, aft_state, bfr_amt, aft_amt, timestamp
BR_DATA Donnees actuelles des records acct_id, record_id, load_id, state, amount, cs_flag
BRD_EU_JC_ITEMS Balance Carree (JC) period_JC, acct_id, record_id, amount

🔢 Types d'Operations dans BR_AUDIT

TYPE Signification Description
0 STATE_CHANGE Changement d'etat (ex: 3→4 = reconciliation)
1 LOAD Chargement de fichier (import)
5 BALANCE Mise a jour du solde
15 BALANCE_UPDATE Mise a jour cumulative du solde (bfr_amt → aft_amt)
18 DEBUT_ROLLBACK Initialisation du rollback (precede TYPE=16)
16 EXEC_ROLLBACK Execution du rollback (suppression effective)
🚨
TYPE=18 puis TYPE=16 = ROLLBACK FICHIER
Ces deux types vont toujours ensemble avec le meme WHICHONE (identifiant du chargement).
TYPE=18 initie le rollback, TYPE=16 l'execute.

🔴 Probleme Identifie

Le curseur Curseur_Attendus_DC_Max (lignes 471-629) calcule DC_MAX 2 fois par ligne via une sous-requete correlee :

-- PROBLEME : Cette sous-requete est executee pour CHAQUE ligne
-- Une fois dans le SELECT :
(SELECT MAX(E.TRANSACTION_DATE)
 FROM BRR_TRANSACTIONS E
 WHERE E.RECONCILIATION_REFERENCE = BRR_TRANSACTIONS.RECONCILIATION_REFERENCE
   AND E.ACCOUNT_ID = BRR_TRANSACTIONS.ACCOUNT_ID) as DC_MAX

-- Et une fois dans le WHERE :
AND (SELECT MAX(E.TRANSACTION_DATE)
     FROM BRR_TRANSACTIONS E
     WHERE E.RECONCILIATION_REFERENCE = BRR_TRANSACTIONS.RECONCILIATION_REFERENCE
       AND E.ACCOUNT_ID = BRR_TRANSACTIONS.ACCOUNT_ID) >= TO_DATE(s_DateArrete, 'DD/MM/RRRR')

-- IMPACT : Millions d'executions de la meme sous-requete !
-- CAUSE : Erreur ORA-01555 "Snapshot too old"

🟢 Solution : CTE (Common Table Expression)

Calcul de DC_MAX UNE SEULE FOIS dans un CTE, puis jointure :

CURSOR Curseur_Attendus_DC_Max IS
WITH
-- CTE 1 : Pre-calcul DC_MAX (UNE SEULE FOIS pour toutes les transactions)
dc_max_calc AS (
    SELECT /*+ MATERIALIZE */
        RECONCILIATION_REFERENCE,
        ACCOUNT_ID,
        MAX(TRANSACTION_DATE) AS DC_MAX
    FROM BRR_TRANSACTIONS
    WHERE RECONCILIATION_REFERENCE IS NOT NULL
    GROUP BY RECONCILIATION_REFERENCE, ACCOUNT_ID
),
-- CTE 2 : Hierarchie des comptes (evite sous-requetes repetees)
compte_hierarchie AS (
    SELECT /*+ MATERIALIZE */
        H.ACCOUNT_ID,
        B2.ACCT_NUM AS SERVICE,
        RPAD(B3.ACCT_NUM, LENGTH(B3.ACCT_NUM)-2) AS TYPE_RAPRO_CODE,
        DECODE(RPAD(B3.ACCT_NUM, LENGTH(B3.ACCT_NUM)-2),
            'RB', 'Rapprochement Bancaire',
            'RC', 'Rapprochement de Controle',
            'JC', 'Justification de compte',
            'CL', 'Compte de liaison') AS TYPE_RAPRO
    FROM BRR_ACCOUNT_HIERARCHIES H
    JOIN BS_ACCTS B2 ON B2.ACCT_ID = H.LEVEL_02_ACCOUNT_ID
    JOIN BS_ACCTS B3 ON B3.ACCT_ID = H.LEVEL_03_ACCOUNT_ID
)
-- Requete OUTSTANDING
SELECT /*+ LEADING(T A CH CM) USE_HASH(A CH CM) */ DISTINCT ...
FROM BRR_TRANSACTIONS T
JOIN compte_hierarchie CH ON CH.ACCOUNT_ID = T.ACCOUNT_ID
WHERE T.STATE = 'OUTSTANDING'

UNION ALL  -- Pas de UNION car OUTSTANDING/RECONCILED sont mutuellement exclusifs

-- Requete RECONCILED avec DC_MAX pre-calcule
SELECT /*+ LEADING(T A CH CM DCM) USE_HASH(A CH CM DCM) */ DISTINCT ...
FROM BRR_TRANSACTIONS T
JOIN compte_hierarchie CH ON CH.ACCOUNT_ID = T.ACCOUNT_ID
JOIN dc_max_calc DCM ON DCM.RECONCILIATION_REFERENCE = T.RECONCILIATION_REFERENCE
                     AND DCM.ACCOUNT_ID = T.ACCOUNT_ID
WHERE T.STATE = 'RECONCILED'
  AND DCM.DC_MAX >= TO_DATE(s_DateArrete, 'DD/MM/RRRR')  -- Filtre sur DC_MAX pre-calcule
